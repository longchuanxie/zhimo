# 知识库按需调研实施计划

## 一、任务概述

为项目级知识库增加「按需调研」能力，形成闭环：

```text
知识提取（文档/对话/资料 → 知识草稿）
  ↓
知识结构化（类型枚举 + 状态流转 + 版本演进）
  ↓
按需调研（Agent 通过 Tool Use 动态查询知识库）
  ↓
提升问答准确性（不再依赖静态全量注入）
```

用户已确认两项关键决策：
1. **按需调研方案**：Tool Use 动态查询（最重量级方案，Agent 在对话中主动调用工具检索知识）
2. **知识来源**：三种全覆盖
   - 从文档自动提取（保存为草稿，事后审阅）
   - 从对话自动提取（保存为草稿，事后审阅）
   - 从资料半自动提取（弹窗预览草稿，确认后保存）

#### 2.1 新建 `src/services/agent/knowledgeTools.ts`

```ts
import type { ToolDefinition, ToolExecutor } from '@/types'
import { listKnowledge } from '@/services/knowledge/KnowledgeService'

/// search_knowledge 工具定义
export const SEARCH_KNOWLEDGE_TOOL: ToolDefinition = {
  type: 'function',
  function: {
    name: 'search_knowledge',
    description: '在当前项目知识库中按关键词搜索已确认的知识条目。用于回答用户问题前检索项目设定、角色、世界观、规则等。',
    parameters: {
      type: 'object',
      properties: {
        keywords: { type: 'array', items: { type: 'string' }, description: '搜索关键词列表' },
        type: { type: 'string', description: '知识类型过滤：character/setting/worldview/plot/rule/fact' },
        limit: { type: 'integer', description: '最大返回数量，默认 5', default: 5 },
      },
      required: ['keywords'],
    },
  },
}

/// 构造知识库查询工具执行器（闭包绑定 projectId）
export function createKnowledgeToolExecutors(projectId: string): Map<string, ToolExecutor> {
  const executors = new Map<string, ToolExecutor>()
  executors.set('search_knowledge', async (args) => {
    const keywords = (args.keywords as string[]) ?? []
    const type = args.type as string | undefined
    const limit = (args.limit as number) ?? 5
    const result = await listKnowledge(projectId, 'confirmed')
    if (!result.ok) return JSON.stringify({ error: result.error.message })
    let items = result.data
    if (type) items = items.filter((k) => k.type === type)
    // 关键词匹配（标题 + 内容 + 摘要，大小写不敏感）
    const lower = keywords.map((k) => k.toLowerCase())
    const matched = items
      .filter((k) => lower.some(
        (kw) => k.title.toLowerCase().includes(kw)
          || k.content.toLowerCase().includes(kw)
          || (k.summary ?? '').toLowerCase().includes(kw),
      ))
      .slice(0, limit)
      .map((k) => ({
        id: k.id, title: k.title, type: k.type,
        summary: k.summary, content: k.content,
        confidence: k.confidence,
      }))
    return JSON.stringify({ items: matched })
  })
  return executors
}

/// 答题任务的可用工具列表
export const ANSWER_QUESTION_TOOLS = [SEARCH_KNOWLEDGE_TOOL]
```

```ts
  : new Map()

  callResult.ok &&
  callResult.data.toolCalls &&
  callResult.data.toolCalls.length > 0 &&
  toolRounds < MAX_TOOL_ROUNDS

    let resultContent: string
    if (!executor) {
      resultContent = JSON.stringify({ error: `未知工具：${tc.function.name}` })
    } else {
      try {
        const args = JSON.parse(tc.function.arguments) as Record<string, unknown>
        resultContent = await executor(args)
      } catch (e) {
        resultContent = JSON.stringify({ error: '工具执行失败', detail: String(e) })
      }
    }
    workingMessages.push({
      role: 'tool',
      content: resultContent,
      toolCallId: tc.id,
    })
  }
  // 再次调用模型，带工具结果
  callResult = await invokeModel(workingMessages, input.signal)
  if (!callResult.ok) break
}

// 最终解析（callResult.data.content 为最终回复）
```

### 阶段 4：知识自动提取

#### 4.1 新建 `src/services/knowledge/KnowledgeExtractor.ts`

```ts
import type { ModelMessage } from '@/types'
import { callModel, callModelDirect, getConfigByTask, getEnabledProvider } from '@/services/model/ModelService'
import type { KnowledgeType } from '@/constants/knowledgeTypes'

export type KnowledgeDraft = {
  title: string
  type: KnowledgeType
  content: string
  summary: string
  confidence: number
  sourceType: 'document' | 'conversation' | 'source'
  sourceId: string
}

const EXTRACT_PROMPT = `你是一个知识抽取助手。从以下内容中提取可作为长期创作参考的知识条目。
仅提取明确的事实/设定/规则，不要编造。每条知识包含：
- title: 简洁标题（≤30字）
- type: character/setting/worldview/plot/rule/fact 之一
- content: 详细内容
- summary: 一句话概括
- confidence: 0~1 置信度

输出严格 JSON：{ "items": [...] }
若无可用知识，输出 { "items": [] }。`

async function invokeExtraction(messages: ModelMessage[]): Promise<KnowledgeDraft[]> {
  // 复用 chat 任务配置，失败回退到第一个启用的 provider
  const configResult = await getConfigByTask('chat')
  let rawContent: string
  if (configResult.ok && configResult.data) {
    const r = await callModel({ modelConfigId: configResult.data.id, messages, temperature: 0 })
    if (!r.ok) throw new Error(r.error.message)
    rawContent = r.data.content
  } else {
    const providerResult = await getEnabledProvider()
    if (!providerResult.ok || !providerResult.data) throw new Error('未配置模型')
    const r = await callModelDirect({
      provider: providerResult.data,
      modelName: providerResult.data.defaultModelName,
      messages, temperature: 0,
    })
    if (!r.ok) throw new Error(r.error.message)
    rawContent = r.data.content
  }
  // 解析 JSON（容错：提取第一个 { 到最后一个 }）
  const start = rawContent.indexOf('{')
  const end = rawContent.lastIndexOf('}')
  if (start < 0 || end < 0) return []
  const parsed = JSON.parse(rawContent.slice(start, end + 1)) as { items: KnowledgeDraft[] }
  return parsed.items ?? []
}

/// 从文档提取知识（自动，保存为草稿）
export async function extractFromDocument(params: {
  projectId: string
  documentId: string
  documentTitle: string
  documentContent: string
}): Promise<KnowledgeDraft[]> {
  const messages: ModelMessage[] = [
    { role: 'system', content: EXTRACT_PROMPT },
    { role: 'user', content: `文档标题：${params.documentTitle}\n\n文档内容：\n${params.documentContent.slice(0, 8000)}` },
  ]
  const items = await invokeExtraction(messages)
  return items.map((it) => ({ ...it, sourceType: 'document', sourceId: params.documentId }))
}

/// 从对话提取知识（自动，保存为草稿）
export async function extractFromConversation(params: {
  projectId: string
  threadId: string
  messages: Array<{ role: string; content: string }>
}): Promise<KnowledgeDraft[]> {
  const transcript = params.messages
    .slice(-20) // 最近 20 条
    .map((m) => `${m.role === 'user' ? '用户' : '助手'}：${m.content}`)
    .join('\n\n')
  const messages: ModelMessage[] = [
    { role: 'system', content: EXTRACT_PROMPT },
    { role: 'user', content: `以下是对话记录，提取其中明确的事实/设定/规则：\n\n${transcript}` },
  ]
  const items = await invokeExtraction(messages)
  return items.map((it) => ({ ...it, sourceType: 'conversation', sourceId: params.threadId }))
}

/// 从资料提取知识（半自动，UI 预览后保存）
export async function extractFromSource(params: {
  projectId: string
  sourceId: string
  sourceTitle: string
  sourceContent: string
}): Promise<KnowledgeDraft[]> {
  const messages: ModelMessage[] = [
    { role: 'system', content: EXTRACT_PROMPT },
    { role: 'user', content: `资料标题：${params.sourceTitle}\n\n资料内容：\n${params.sourceContent.slice(0, 8000)}` },
  ]
  const items = await invokeExtraction(messages)
  return items.map((it) => ({ ...it, sourceType: 'source', sourceId: params.sourceId }))
}
```

## 四、假设与决策

1. **工具调用循环上限 3 轮**：防止死循环；正常场景 1-2 轮即可收敛。
2. **仅 `answer_question` 任务启用工具**：改写/扩写等任务保持静态注入，避免延迟。后续可扩展。
3. **静态知识注入降为 2 条**：作为基线，详细内容由 Tool Use 检索。平衡延迟与准确性。
4. **知识类型 6 种**：character/setting/worldview/plot/rule/fact，覆盖小说场景核心。旧数据 `normalizeKnowledgeType` 归为 fact。
5. **文档/对话自动提取**：保存为 `pending` 草稿，不直接确认，避免噪声污染。用户在知识库审阅。
6. **资料半自动提取**：弹窗预览，用户勾选保存，更可控。
7. **对话提取阈值 ≥4 条消息**：避免单条问答就触发提取。
8. **版本管理**：`replaceKnowledge` 创建新版本，旧版本 `deprecated` + `replacedById`，不物理删除。
9. **工具执行错误不中断**：工具失败时返回错误 JSON 给模型，模型可继续回复或说明失败。
10. **模型兼容性**：Tool Use 需模型支持 OpenAI function calling。不支持时模型忽略 tools 正常回复，不影响现有流程。
