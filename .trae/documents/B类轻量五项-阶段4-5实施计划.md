```ts
/// 模型列表缓存:避免同一服务商短时间内重复请求 /v1/models
/// TTL 5 分钟,过期后下次查询自动刷新
const modelListCache = new Map<string, { models: ModelInfo[]; ts: number }>()
const MODEL_LIST_CACHE_TTL = 5 * 60 * 1000

/// 带缓存的模型列表查询
/// 命中且未过期直接返回缓存;否则调 listProviderModels 并写入缓存
/// 远程失败或返回空时不写入缓存(保证下次仍会尝试拉取)
async function fetchModelsCached(pid: string): Promise<ServiceResult<ModelInfo[]>> {
  const cached = modelListCache.get(pid)
  if (cached && Date.now() - cached.ts < MODEL_LIST_CACHE_TTL) {
    return ok(cached.models)
  }
  const result = await listProviderModels(pid)
  if (result.ok && result.data.length > 0) {
    modelListCache.set(pid, { models: result.data, ts: Date.now() })
  }
  return result
}
```

**依赖**:需补充 import `ok` from `@/types/service`(ServiceResult 构造器)

#### 5.2.3 新增 renameThread(公开 API)

**位置**:sendMessage 函数之前(约 367 行之前)

```ts
/// 重命名对话线程
/// @internal autoRenameThreadIfNeeded 内部调用,也供 UI 手动改名使用
export async function renameThread(
  threadId: EntityId,
  title: string,
): Promise<ServiceResult<AgentThread>> {
  try {
    const trimmed = title.trim()
    if (!trimmed) {
      return err({
        code: VALIDATION_ERROR,
        message: '标题不能为空',
        retryable: false,
      })
    }
    // 截断到 12 字(按 Array.from 处理 Unicode)
    const truncated = Array.from(trimmed).slice(0, 12).join('')

    const existing = await findThreadById(threadId)
    if (!existing) {
      return err({
        code: AGENT_THREAD_NOT_FOUND,
        message: '没有找到助手对话',
        retryable: false,
      })
    }
    await updateThreadTitle(threadId, truncated)
    const updated = await findThreadById(threadId)
    if (!updated) {
      return err({
        code: 'UNKNOWN_ERROR',
        message: '线程更新后查询失败',
        retryable: true,
      })
    }
    return ok(updated)
  } catch (error) {
    return err(fromUnknown(error))
  }
}
```

#### 5.2.4 新增 autoRenameThreadIfNeeded(内部 fire-and-forget)

**位置**:renameThread 之后

```ts
/// 首回合后自动命名会话
///
/// 触发条件:
/// - thread.messageCount === 0(sendMessage 开始时查询的原始值,表示首回合)
/// - thread.title 以"新对话"开头(前端创建时的默认格式)
///
/// 行为:
/// - 调用 LLM 生成 ≤12 字标题,成功则更新
/// - LLM 失败或未配置模型 → 回退截取用户消息前 20 字
/// - 整个函数不抛错,失败时静默回退
///
/// @internal 由 sendMessage fire-and-forget 调用,不阻塞主流程
export async function autoRenameThreadIfNeeded(
  thread: AgentThread,
  userMessage: AgentMessage,
  _assistantMessage: AgentMessage,
): Promise<void> {
  // 1. 触发条件判定
  if (thread.messageCount !== 0) return
  if (!thread.title.startsWith('新对话')) return

  const fallbackTitle = Array.from(userMessage.content.trim()).slice(0, 20).join('')

  try {
    // 2. 获取模型配置(复用 sendMessage 的回退逻辑)
    const configResult = await getConfigByTask('chat')
    if (!configResult.ok) {
      await renameThread(thread.id, fallbackTitle).catch(() => {})
      return
    }

    let generatedTitle: string

    if (configResult.data) {
      const result = await callModel({
        modelConfigId: configResult.data.id,
        messages: [{
          role: 'user',
          content: AUTO_TITLE_PROMPT.replace('{content}', userMessage.content),
        }],
        temperature: 0,
        maxOutputTokens: 64,
      })
      if (!result.ok) {
        await renameThread(thread.id, fallbackTitle).catch(() => {})
        return
      }
      generatedTitle = result.data.content.trim()
    } else {
      // 回退到第一个启用的服务商
      const providerResult = await getEnabledProvider()
      if (!providerResult.ok || !providerResult.data) {
        await renameThread(thread.id, fallbackTitle).catch(() => {})
        return
      }
      const result = await callModelDirect({
        provider: providerResult.data,
        modelName: providerResult.data.defaultModelName,
        messages: [{
          role: 'user',
          content: AUTO_TITLE_PROMPT.replace('{content}', userMessage.content),
        }],
        temperature: 0,
      })
      if (!result.ok) {
        await renameThread(thread.id, fallbackTitle).catch(() => {})
        return
      }
      generatedTitle = result.data.content.trim()
    }

    // 3. LLM 返回空内容时回退
    if (!generatedTitle) {
      await renameThread(thread.id, fallbackTitle).catch(() => {})
      return
    }

    // 4. 更新标题(renameThread 内部会截断到 12 字)
    await renameThread(thread.id, generatedTitle)
  } catch {
    // 任何异常都静默回退
    await renameThread(thread.id, fallbackTitle).catch(() => {})
  }
}
```

#### 5.2.5 sendMessage 插入 fire-and-forget 调用

**位置**:[sendMessage 第 619 行 `return ok(...)` 之前](file:///d:/workplace/idea/zhimo/src/services/agent/AgentService.ts#L619)


**关键说明**:
- `thread` 是 sendMessage 开始时(375 行)查询的原始对象,`thread.messageCount === 0` 表示该线程此前无消息,即首回合
- `bumpThreadMessageStats`(449、592 行)只更新 DB 中的 message_count,不刷新 thread 对象,所以判定条件正确
- fire-and-forget 模式:即使 autoRenameThreadIfNeeded 失败,也不影响 sendMessage 返回;前端 `loadThreads()` 会拉到旧标题,但下次刷新会拉到新标题(数秒延迟可接受)

## 假设与决策

1. **缓存 TTL 5 分钟**:平衡新鲜度与请求成本,用户编辑服务商基本信息时不会刷新缓存(仅模型列表相关),可接受
2. **TEMP_DEFAULT_MODEL 保留 'gpt-4o-mini'**:不改空串,避免改动 Service 层校验;UI 已强制第二步选模型,占位值不持久化
3. **触发条件用 thread.messageCount === 0**:`thread` 是 sendMessage 开始时查询的原始对象,首回合时 messageCount=0。规划文档原文"messageCount === 1"是笔误(那是指 bumpThreadMessageStats 后的 DB 值,不是 thread 对象值)
4. **autoRenameThreadIfNeeded 导出**:虽然主要是内部使用,但导出后便于单元测试直接调用,避免通过 sendMessage 全链路测试
5. **LLM 调用用 chat 任务配置**:与主对话使用相同模型,避免引入额外配置项;temperature=0 保证标题稳定
6. **fire-and-forget 不阻塞 sendMessage**:前端 `loadThreads()` 立即拉到旧标题,数秒后下次交互拉到新标题,用户体验可接受
